# to build the image

# run the image in dev mode (mount local workspace)
docker run -it --rm   --net=host  -v $(pwd):/app   ros2-jazzy-dev

# run the image in deploy mode (files are local to image)
docker run -it --rm   --net=host     ros2-jazzy-dev
